Sub Random()
Dim RNG As Range
    Set RNG = Range("A1:J20")

Dim randomCell As Long
Dim i As Integer
Dim n As Integer

n = 20
    
    For i = 1 To n
    
    Do
    randomCell = Int(Rnd * RNG.Cells.Count) + 1
    Loop While RNG.Cells(randomCell).Interior.ColorIndex <> xlNone
    RNG.Cells(randomCell).Interior.Color = vbYellow

    Next i

End Sub
